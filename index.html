async function sendMessage(){ const text = textarea.value.trim(); if (!text && !pendingB64) return; const isPremium = currentUser?.premium; if (!isPremium) { const u = getUsage(); if (u.msgs >= FREE_MSGS) { updateLimitBadge(); return; } if (pendingB64 && u.imgs >= FREE_IMGS) { alert(`Limite de ${FREE_IMGS} images atteinte !`); return; } u.msgs++; if (pendingB64) u.imgs++; saveUsage(u); updateLimitBadge(); } const imgSrc = pendingB64 ? `data:${pendingType};base64,${pendingB64}` : null; const userText = text || 'Décris cette image.'; textarea.value = ''; textarea.style.height = 'auto'; addUserMsg(userText, imgSrc); // On garde l'historique simple pour Groq history.push({role:'user', content: userText}); removeImage(); $('send-btn').disabled = true; showTyping(); $('chat-err').style.display = 'none'; try { const res = await fetch('/api/chat', { method:'POST', headers:{'Content-Type':'application/json'}, body: JSON.stringify({ messages: history, userId: currentUser?.id }) }); if (!res.ok) throw new Error(`Erreur serveur ${res.status}`); const data = await res.json(); if (data.error) throw new Error(data.error); // ✅ NOUVEAU FORMAT GROQ const reply = data.reply || "Désolé, pas de réponse."; history.push({role:'assistant', content: reply}); hideTyping(); addAiMsg(reply); // Sauvegarder const title = history.find(m => m.role === 'user')?.content || 'Chat'; fetch('/api/chats', { method:'POST', headers:{'Content-Type':'application/json'}, body: JSON.stringify({action:'save', user_id:currentUser.id, chat_id:currentChatId, title:title.substring(0,40), messages:history}) }).then(() => loadChatHistory()).catch(() => {}); } catch(e) { hideTyping(); $('chat-err').textContent = '⚠️ ' + e.message; $('chat-err').style.display = 'block'; } $('send-btn').disabled = false; textarea.focus(); }
